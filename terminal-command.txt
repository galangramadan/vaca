Create Model
    Books:
    sequelize model:generate --name books --attributes title:string,image:string,author:string,description:text,category_id:integer
    Categories:
    sequelize model:generate --name categories --attributes name:string,image:string
    Users:
    sequelize model:generate --name users --attributes email:string,password:string,role:enum:'{admin,user}',phone_number:integer,name:string,gender:enum:'{male,female}',date_of_birth:date,address:text,avatar:string


Migrate
    sequelize db:migrate


Seeders
    Create:
    sequelize seed:generate --name demo-tableName

    Run:
    sequelize db:seed:all
    sequelize db:seed --seed seederfilename.js

    Undo:
    sequelize db:seed:undo:all
